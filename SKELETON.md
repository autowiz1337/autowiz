Quality Score Calculation Function
javascript
const calculateQualityScore = (formData) => {
  let score = 0;
  
  // Required fields (40%)
  const requiredComplete = /* check all required fields */;
  if (requiredComplete) score += 40;
  
  // Standout features (15%)
  if (formData.condition.standoutExterior && 
      formData.condition.standoutInterior) score += 15;
  
  // Premium features (15%)
  const totalFeatures = Object.values(formData.actionDrivers.premiumFeatures)
    .flat().length;
  if (totalFeatures >= 5) score += 15;
  
  // Wow factor (10%)
  if (formData.actionDrivers.wowFactor) score += 10;
  
  // Urgency triggers (10%)
  if (formData.actionDrivers.urgencyTriggers.length > 0) score += 10;
  
  // Special offer (10%)
  if (formData.actionDrivers.specialOffer || 
      formData.actionDrivers.noSpecialOffer) score += 10;
  
  return Math.min(score, 100);
};
